<script>

let steps = [false,false];

let started = false;


function openLink(type){

  if(type === "snap"){

    window.open("https://www.snapchat.com/add/dramani_photo","_blank");

  }

  if(type === "tiktok"){

    window.open("https://www.tiktok.com/@dr.amani92","_blank");

  }

}


function confirmStep(i,btn){

  steps[i] = true;

  btn.classList.add("done");

  btn.innerText = "✔ تم";


  if(steps.every(x=>x) && !started){

    startTimer();

  }

}


function startTimer(){

  started = true;

  let t = 5;

  let el = document.getElementById("timer");


  let interval = setInterval(()=>{

    el.innerText = "جاري فتح الرابط... " + t + " ثواني";

    t--;


    if(t < 0){

      clearInterval(interval);

      el.innerText = "تم فتح الحاسبة ✅";

      document.getElementById("unlock").style.display = "block";

    }

  },1000);

}
